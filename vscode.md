```bash
#------------------- Explorer/Files-section ---------------------------
Cntr + E  # move to files/explorer-section
j   # move downward 
k   # move upward

#--------------------- Terminal --------------------------------------
Cntr + `    # move to terminal


#--------------------- Text Editor ----------------------------------
Cntr + 1    # move to text editor
Cntr + p    # search files by name in current directory
Cntr + P    # open command palette
```
